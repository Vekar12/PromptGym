18def extract_raw_events(**context):


19    """Extract events from source Postgres, push to XCom."""


20    hook = PostgresHook(postgres_conn_id="source_db")


21    run_date = context["ds"]


22    sql = f"""


23        SELECT event_id, user_id, event_type, ts, payload


24        FROM raw_events


25        WHERE DATE(ts) = '{run_date}'


26          AND event_type IN ('purchase', 'signup', 'churn')


27    """


28    df = hook.get_pandas_df(sql)


29    logger.info(f"Extracted {len(df)} rows for {run_date}")


30    return df.to_json()
