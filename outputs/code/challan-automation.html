1# Automated challan status checker + bulk downloader


2import requests, json, time


3from pathlib import Path


4from typing import Optional


5from dataclasses import dataclass


6


7API_BASE = "https://echallan.parivahan.gov.in/api/v1"


8RETRY_LIMIT = 3


9SLEEP_BETWEEN = 1.2  # seconds, avoid rate limiting


10


11@dataclass


12class ChallanResult:


13    vehicle_no: str


14    challans_found: int


15    total_amount: float


16    paid: bool


17    error: Optional[str] = None


18


19def check_challan(vehicle_no: str) -> ChallanResult:


20    for attempt in range(RETRY_LIMIT):


21        try:


22            resp = requests.get(


23                f"{API_BASE}/challan/status",


24                params={"regNo": vehicle_no},


25                timeout=10


26            )


27            resp.raise_for_status()


28            data = resp.json()


29            return ChallanResult(


30                vehicle_no=vehicle_no,


31                challans_found=data["count"],


32                total_amount=data["totalAmount"],


33                paid=data["status"] == "PAID",


34            )


35        except Exception as e:


36            if attempt == RETRY_LIMIT - 1:


37                return ChallanResult(vehicle_no, 0, 0, False, str(e))


38            time.sleep(SLEEP_BETWEEN)


39


40def run_batch(vehicle_list: list[str]) -> list[ChallanResult]:


41    results = []


42    for v in vehicle_list:


43        results.append(check_challan(v))


44        time.sleep(SLEEP_BETWEEN)


45    return results
