""" retry_module.py API retry utility with exponential backoff and jitter. Usage: from retry_module import retry_with_backoff, with_retry result = retry_with_backoff(api_client.get_data, file_id="123") @with_retry(max_retries=5) def fetch_document(doc_id: str) -> dict: return api_client.get(doc_id) """ import time import random import logging from typing import Callable, Any, Optional from functools import wraps # ── Constants ───────────────────────────────────────────────────────────────── MAX_RETRIES: int = 3 BASE_DELAY: float = 1.0 # seconds before first retry MAX_DELAY: float = 30.0 # cap to prevent infinite wait JITTER_FACTOR: float = 0.3 # randomness fraction to prevent thundering herd logger = logging.getLogger(__name__) # ── Exceptions ──────────────────────────────────────────────────────────────── class APIRetryExhausted(Exception): """Raised when all retry attempts have been exhausted.""" def __init__(self, attempts: int, last_error: Exception): self.attempts = attempts self.last_error = last_error super().__init__(f"API call failed after {attempts} attempts: {last_error}") # ── Core Function ───────────────────────────────────────────────────────────── def retry_with_backoff( fn: Callable, *args: Any, max_retries: int = MAX_RETRIES, base_delay: float = BASE_DELAY, retryable_exceptions: tuple = (ConnectionError, TimeoutError), **kwargs: Any ) -> Any: """ Execute fn with exponential backoff retry on failure. Args: fn: Callable to execute max_retries: Maximum number of retry attempts base_delay: Initial delay in seconds (doubles each attempt) retryable_exceptions: Only retry on these exception types """ last_error: Optional[Exception] = None for attempt in range(max_retries + 1): try: return fn(*args, **kwargs) # Attempt the call except retryable_exceptions as e: last_error = e if attempt == max_retries: break # No more retries — raise below # Exponential backoff: 1s, 2s, 4s, 8s... delay = min(base_delay * (2 ** attempt), MAX_DELAY) # Add jitter to prevent all clients retrying simultaneously jitter = random.uniform(-JITTER_FACTOR * delay, JITTER_FACTOR * delay) sleep_time = max(0, delay + jitter) logger.warning(f"Attempt {attempt + 1} failed: {e}. Retrying in {sleep_time:.2f}s...") time.sleep(sleep_time) raise APIRetryExhausted(attempts=max_retries + 1, last_error=last_error) # ── Decorator Version ────────────────────────────────────────────────────────── def with_retry(max_retries: int = MAX_RETRIES, base_delay: float = BASE_DELAY): """Decorator to add retry logic to any function.""" def decorator(fn: Callable) -> Callable: @wraps(fn) # Preserve original function metadata def wrapper(*args: Any, **kwargs: Any) -> Any: return retry_with_backoff(fn, *args, max_retries=max_retries, base_delay=base_delay, **kwargs) return wrapper return decorator # ── Usage Example ───────────────────────────────────────────────────────────── if __name__ == "__main__": # Direct usage def unstable_api_call(file_id: str) -> dict: return {"status": "ok", "file_id": file_id} result = retry_with_backoff(unstable_api_call, file_id="ILNS-2024-4421") # Decorator usage @with_retry(max_retries=5) def fetch_challan_data(bus_id: str) -> dict: return api_client.get(f"/challans/{bus_id}")
